# Variables de entorno para configuración crítica
# Copia este archivo a .env y completa los valores

# API Key de IMDB (obtener en https://imdb-api.com/api)

# Token del bot de Telegram (obtener de @BotFather)

# ID del canal de Telegram (formato: @channel_name o -1001234567890)
TELEGRAM_CHANNEL_ID=your_telegram_channel_id_here

# Credenciales de PLEX Media Server (opcional)
# Opción 1: Usuario y contraseña (método legacy, puede no funcionar)

PLEX_PASS=your_plex_password_here

# Opción 2: Token de acceso (recomendado)
# Obtén tu token en: PLEX → Configuración → Red → Acceso remoto
PLEX_TOKEN=your_plex_token_here

# Configuración del servidor PLEX
# IP o nombre del servidor donde está PLEX (ej: 192.168.1.136, diskstation, localhost)
