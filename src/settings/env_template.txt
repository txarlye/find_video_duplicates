# Variables de entorno para configuración crítica
# Copia este archivo a .env y completa los valores

# API Key de IMDB (obtener en https://imdb-api.com/api)

# Token del bot de Telegram (obtener de @BotFather)

# Token de Plex (obtener desde la configuración de Plex)
PLEX_TOKEN=your_plex_token_here

# URL del servidor Plex (opcional, por defecto localhost:32400)
PLEX_URL=http://localhost:32400
